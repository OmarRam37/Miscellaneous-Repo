# Imputar edad
def imputar_edad(cols):
    Age = cols [0]
    Pclass = cols [1]
    
    if pd.isna(Age):
        if Pclass == 1:
            return 37
        elif Pclass == 2: 
            return 29
        else:
            return 24
    else:
        return Age
